class Solution{
  public:
    int countSubArrayProductLessThanK(const vector<int>& a, int n, long long k) {
        long long int start=0,end=0,count=0,product=1;
       while(end<n){
           product *= a[end];
           while(start<n and product>=k){
               product = product/a[start];
               start++;
           }
           if(product<k)
           count += end-start+1;
           end++;
       }
       return count;
    }
};
